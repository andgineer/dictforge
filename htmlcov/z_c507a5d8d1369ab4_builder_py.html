514 def _prepare_combined_entries(self, in_lang: str, out_lang: str) -> tuple[Path, int]: # noqa: C901 

515 """Aggregate entries from each configured source, merging senses/examples by word.""" 

516 if len(self._sources) == 1: 

517 source = self._sources[0] 

518 result = source.get_entries(in_lang, out_lang) 

519 self._log_source_stats(source, in_lang) 

520 return result 

521 

522 combined_dir = self.cache_dir / "combined" 

523 combined_dir.mkdir(parents=True, exist_ok=True) 

524 source_tag = "_".join(type(src).__name__ for src in self._sources) 

525 source_tag_slug = self._slugify(source_tag) 

526 filename = f"{self._slugify(in_lang)}__{self._slugify(out_lang)}__{source_tag_slug}.jsonl" 

527 combined_path = combined_dir / filename 

528 

529 merged_entries: OrderedDict[str, dict[str, Any]] = OrderedDict() 

530 for source in self._sources: 

531 data_path, _ = source.get_entries(in_lang, out_lang) 

532 self._log_source_stats(source, in_lang) 

533 try: 

534 with data_path.open("r", encoding="utf-8") as fh: 

535 for line in fh: 

536 payload = line.strip() 

537 if not payload: 

538 continue 

539 try: 

540 entry = json.loads(payload) 

541 except json.JSONDecodeError as exc: 

542 raise KaikkiParseError(data_path, exc) from exc 

543 if not entry_has_content(entry): 

544 continue 

545 word = entry.get("word") 

546 if not isinstance(word, str): 

547 continue 

548 key = word.lower() 

549 if key not in merged_entries: 

550 merged_entries[key] = copy.deepcopy(entry) 

551 else: 

552 self._merge_entry(merged_entries[key], entry) 

553 except OSError as exc: 

554 raise KaikkiDownloadError( 

555 f"Failed to read source dataset '{data_path}': {exc}", 

556 ) from exc 

557 

558 if not merged_entries: 

559 raise KaikkiDownloadError( 

560 f"No entries produced by configured sources for {in_lang} → {out_lang}.", 

561 ) 

562 

563 with combined_path.open("w", encoding="utf-8") as dst: 

564 for entry in merged_entries.values(): 

565 dst.write(json.dumps(entry, ensure_ascii=False) + "\n") 

566 

567 return combined_path, len(merged_entries)

569 def _log_source_stats(self, source: DictionarySource, language: str) -> None: 

570 """Print a one-time summary of filtered entries for ``language``.""" 

571 key = (id(source), language.lower()) 

572 if key in self._logged_source_stats: 

573 return 

574 stats_getter = getattr(source, "get_filter_stats", None) 

575 if not callable(stats_getter): 

576 return 

577 stats = stats_getter(language) 

578 if not stats: 

579 return 

580 count = stats.get("count") 

581 if count is None: 

582 return 

583 matched = stats.get("matched_entries", count) 

584 skipped = stats.get("skipped_empty") 

585 skipped_label = f", skipped {skipped:,} empty" if skipped is not None else "" 

586 self._console.print( 

587 (f"[dictforge] {language}: kept {count:,} of {matched:,} entries{skipped_label}"), 

588 style="cyan", 

589 ) 

590 self._logged_source_stats.add(key)

592 def _merge_entry(self, target: dict[str, Any], incoming: dict[str, Any]) -> None: 

593 """Combine senses/examples from ``incoming`` into ``target`` without duplicates.""" 

594 target_senses = target.get("senses") 

595 incoming_senses = incoming.get("senses") 

596 if not isinstance(target_senses, list) or not isinstance(incoming_senses, list): 

597 return 

598 

599 index: dict[tuple[str, ...], dict[str, Any]] = {} 

600 for sense in target_senses: 

601 if not isinstance(sense, dict): 

602 continue 

603 glosses = sense.get("glosses") 

604 if isinstance(glosses, list) and glosses: 

605 key = tuple(str(g) for g in glosses) 

606 index[key] = sense 

607 

608 for sense in incoming_senses: 

609 if not isinstance(sense, dict): 

610 continue 

611 glosses = sense.get("glosses") 

612 if isinstance(glosses, list) and glosses: 

613 key = tuple(str(g) for g in glosses) 

614 self._merge_examples(index[key], sense) 

615 else: 

616 target_senses.append(copy.deepcopy(sense))

618 def _merge_examples(self, target_sense: dict[str, Any], incoming_sense: dict[str, Any]) -> None: 

619 """Append new example blocks from ``incoming_sense`` onto ``target_sense``.""" 

620 incoming_examples = incoming_sense.get("examples") 

621 if not isinstance(incoming_examples, list) or not incoming_examples: 

622 return 

623 

624 target_examples = target_sense.get("examples") 

625 if not isinstance(target_examples, list): 

626 target_examples = [] 

627 target_sense["examples"] = target_examples 

628 

629 for example in incoming_examples: 

630 exemplar = copy.deepcopy(example) 

631 if exemplar not in target_examples: 

632 target_examples.append(exemplar)

645 def _kindle_lang_code(self, code: str | None, override: str | None = None) -> str: 

646 """Derive the Kindle language identifier, applying overrides/fallbacks.""" 

647 if override: 

648 normalized_override = override.lower() 

649 if normalized_override in KINDLE_SUPPORTED_LANGS: 

650 return normalized_override 

651 raise KindleBuildError( 

652 ( 

653 f"Kindle language override '{override}' is not supported by Kindle. " 

654 "Check the supported list and pick a valid code." 

655 ), 

656 ) 

657 

658 if not code: 

659 return "en" 

660 

661 normalized = code.lower() 

662 if normalized in KINDLE_SUPPORTED_LANGS: 

663 return normalized 

664 

665 overrides = { 

666 "sr": "hr", 

667 "en": "en-us", 

668 } 

669 normalized = overrides.get(normalized, normalized) 

670 

671 if normalized == "en": 

672 return "en-us" 

673 

674 return normalized if normalized in KINDLE_SUPPORTED_LANGS else "en"

711 raise 

712 else: 

713 db_capture.finish() 

714 finally: 

715 db_capture.stop() 

716 mobi_base = outdir / f"{in_lang}-{out_lang}" 

717 shutil.rmtree(mobi_base, ignore_errors=True) 

718 kindle_capture = _KindleProgressCapture( 

719 console=self._console, 

720 enabled=self._show_progress, 

721 total_hint=entry_count if entry_count else None, 

722 ) 

723 kindle_capture.start() 

724 fallback_exc: FileNotFoundError | None = None 

725 try: 

726 with redirect_stdout(kindle_capture), redirect_stderr(kindle_capture): # type: ignore 

727 dc.export_to_kindle( 

728 kindlegen_path=kindlegen_path, 

729 try_to_fix_failed_inflections=try_fix_inflections, # type: ignore[arg-type] # bug in the lib 

730 author="Wiktionary via Wiktextract (Kaikki.org)", 

731 title=title, 

732 mobi_temp_folder_path=str(mobi_base), 

733 mobi_output_file_path=f"{mobi_base}.mobi", 

734 ) 

735 except FileNotFoundError as exc: 

736 fallback_exc = exc 

737 except Exception: 

738 self._emit_creator_output("Kindle export output", kindle_capture) 

739 raise 

740 else: 

741 kindle_capture.finish() 

742 finally: 

743 kindle_capture.stop() 

744 

745 if fallback_exc is None: 

746 self._announce_summary(in_lang, out_lang, entry_count, kindle_capture) 

747 return entry_count 

748 

749 opf_path = mobi_base / "OEBPS" / "content.opf" 

750 if not opf_path.exists(): 

751 raise KindleBuildError( 

752 "Kindle Previewer failed and content.opf is missing; see previous output.", 

753 ) from fallback_exc 

754 self._console.print( 

755 "[dictforge] Kindle Previewer fallback: fixing metadata and retrying", 

756 style="yellow", 

757 ) 

758 self._ensure_opf_languages(opf_path, kindle_in, kindle_out, title) 

759 self._run_kindlegen(kindlegen_path, opf_path) 

760 mobi_path = mobi_base / "OEBPS" / "content.mobi" 

761 if not mobi_path.exists(): 

762 raise KindleBuildError( 

763 "Kindle Previewer did not produce content.mobi even after fixing metadata.", 

764 ) from fallback_exc

767 dc.mobi_path = str(final_path) 

768 shutil.rmtree(mobi_base, ignore_errors=True) 

769 

770 self._announce_summary(in_lang, out_lang, entry_count, kindle_capture) 

771 return entry_count

773 def _ensure_opf_languages( # noqa: PLR0912,C901 

774 self, 

775 opf_path: Path, 

776 primary_code: str, 

777 secondary_code: str, 

778 title: str, 

779 ) -> None: 

780 """Patch the OPF metadata so Kindle recognises the dictionary languages.""" 

781 print( 

782 ( 

783 f"[dictforge] Preparing OPF languages: source→'{primary_code}', " 

784 f"target→'{secondary_code}'" 

785 ), 

786 flush=True, 

787 ) 

788 

789 tree = ET.parse(opf_path) 

790 ns = { 

791 "opf": "http://www.idpf.org/2007/opf", 

792 "dc": "http://purl.org/dc/elements/1.1/", 

793 "legacy": "http://purl.org/metadata/dublin_core", 

794 } 

795 ET.register_namespace("", ns["opf"]) 

796 ET.register_namespace("dc", ns["dc"]) 

797 metadata = tree.find("opf:metadata", ns) 

798 if metadata is None: 

799 metadata = ET.SubElement(tree.getroot(), "{http://www.idpf.org/2007/opf}metadata") 

800 

801 # modern dc:title/creator fallbacks 

802 if metadata.find("dc:title", ns) is None: 

803 title_elem = ET.SubElement(metadata, "{http://purl.org/dc/elements/1.1/}title") 

804 title_elem.text = title or "dictforge dictionary" 

805 

806 if metadata.find("dc:creator", ns) is None: 

807 legacy = metadata.find("opf:dc-metadata", ns) 

808 creator_text = None 

809 if legacy is not None: 

810 legacy_creator = legacy.find("legacy:Creator", ns) 

811 if legacy_creator is not None: 

812 creator_text = legacy_creator.text 

813 ET.SubElement(metadata, "{http://purl.org/dc/elements/1.1/}creator").text = ( 

814 creator_text or "dictforge" 

815 ) 

816 

817 # modern dc:language entries 

818 for elem in list(metadata.findall("dc:language", ns)): 

819 metadata.remove(elem) 

820 ET.SubElement(metadata, "{http://purl.org/dc/elements/1.1/}language").text = primary_code 

821 

822 # legacy dc-metadata block 

823 legacy = metadata.find("opf:dc-metadata", ns) 

824 if legacy is not None: 

825 for elem in legacy.findall("legacy:Language", ns): 

826 elem.text = primary_code 

827 if legacy.find("legacy:Title", ns) is None: 

828 ET.SubElement(legacy, "{http://purl.org/metadata/dublin_core}Title").text = title 

829 if legacy.find("legacy:Creator", ns) is None: 

830 ET.SubElement( 

831 legacy, 

832 "{http://purl.org/metadata/dublin_core}Creator", 

833 ).text = "dictforge" 

834 

835 # x-metadata block used by Kindle dictionaries 

836 x_metadata = metadata.find("opf:x-metadata", ns) 

837 if x_metadata is not None: 

838 dict_in = x_metadata.find("opf:DictionaryInLanguage", ns) 

839 if dict_in is not None: 

840 dict_in.text = primary_code 

841 dict_out = x_metadata.find("opf:DictionaryOutLanguage", ns) 

842 if dict_out is not None: 

843 dict_out.text = secondary_code 

844 default_lookup = x_metadata.find("opf:DefaultLookupIndex", ns) 

845 if default_lookup is None: 

846 ET.SubElement( 

847 x_metadata, 

848 "{http://www.idpf.org/2007/opf}DefaultLookupIndex", 

849 ).text = "default" 

850 

851 tree.write(opf_path, encoding="utf-8", xml_declaration=True)

853 def _run_kindlegen(self, kindlegen_path: str, opf_path: Path) -> None: 

854 """Invoke Kindle Previewer/kindlegen and surface helpful errors.""" 

855 if not kindlegen_path: 

856 raise KindleBuildError("Kindle Previewer path is empty; cannot invoke kindlegen.") 

857 

858 process = subprocess.run( 

859 [kindlegen_path, opf_path.name], 

860 check=False, 

861 capture_output=True, 

862 text=True, 

863 cwd=str(opf_path.parent), 

864 ) 

865 if process.returncode != 0: 

866 raise KindleBuildError( 

867 "Kindle Previewer reported an error after fixing metadata:\n" 

868 f"STDOUT:\n{process.stdout}\nSTDERR:\n{process.stderr}", 

869 )

872 self,
